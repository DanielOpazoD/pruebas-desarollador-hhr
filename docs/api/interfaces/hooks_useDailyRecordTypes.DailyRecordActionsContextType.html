    Interface DailyRecordActionsContextType

    Actions-only part of the context. These functions are generally stable (referentially consistent) and won't trigger re-renders when data changes.

    interface DailyRecordActionsContextType {
        addCMA: (data: Omit<CMAData, "id" | "timestamp">) => void;
        addDischarge: AddDischargeAction;
        addTransfer: AddTransferAction;
        canDischargePatient: (patient: PatientData) => boolean;
        canMovePatient: (
            sourceBedId: string,
            targetBedId: string,
            record: DailyRecord | null,
        ) => { canMove: boolean; reason?: string };
        clearAllBeds: () => void;
        clearPatient: (bedId: string) => void;
        confirmMedicalSpecialtyNoChanges: (
            input: {
                actor: Partial<MedicalHandoffActor>;
                comment?: string;
                dateKey?: string;
                specialty: MedicalSpecialty;
            },
        ) => Promise<void>;
        copyPatientToDate: (
            bedId: string,
            targetDate: string,
            targetBedId?: string,
        ) => Promise<void>;
        createDay: (
            copyFromPrevious: boolean,
            specificDate?: string,
            options?: { forceCopyScheduleOverride?: boolean },
        ) => void;
        deleteCMA: (id: string) => void;
        deleteDischarge: DeleteDischargeAction;
        deleteTransfer: DeleteTransferAction;
        ensureMedicalHandoffSignatureLink: (
            scope?: MedicalHandoffScope,
        ) => Promise<ApplicationOutcome<{ handoffUrl: string } | null>>;
        markMedicalHandoffAsSent: (
            doctorName?: string,
            scope?: MedicalHandoffScope,
        ) => Promise<void>;
        moveOrCopyPatient: (
            type: "move" | "copy",
            sourceBedId: string,
            targetBedId: string,
        ) => void;
        refresh: () => void;
        resetDay: () => Promise<void>;
        resetMedicalHandoffState: () => Promise<void>;
        sendMedicalHandoff: (
            templateContent: string,
            targetGroupId: string,
        ) => Promise<void>;
        toggleBedType: (bedId: string) => void;
        toggleBlockBed: (bedId: string, reason?: string) => void;
        toggleExtraBed: (bedId: string) => void;
        undoDischarge: UndoDischargeAction;
        undoTransfer: UndoTransferAction;
        updateBlockedReason: (bedId: string, reason: string) => void;
        updateClinicalCrib: (
            bedId: string,
            field: (keyof PatientData) | "create" | "remove",
            value?: PatientFieldValue,
        ) => void;
        updateClinicalCribCudyr: (
            bedId: string,
            field: keyof CudyrScore,
            value: number,
        ) => void;
        updateClinicalCribCudyrMultiple?: (
            bedId: string,
            fields: Partial<Record<keyof CudyrScore, number>>,
        ) => void;
        updateClinicalCribMultiple: (
            bedId: string,
            fields: Partial<PatientData>,
        ) => void;
        updateCMA: (id: string, updates: Partial<CMAData>) => void;
        updateCudyr: (
            bedId: string,
            field: keyof CudyrScore,
            value: number,
        ) => void;
        updateCudyrMultiple?: (
            bedId: string,
            fields: Partial<Record<keyof CudyrScore, number>>,
        ) => void;
        updateDischarge: UpdateDischargeAction;
        updateHandoffChecklist: (
            shift: "day" | "night",
            field: string,
            value: string | boolean,
        ) => void;
        updateHandoffNovedades: (
            shift: "day" | "night" | "medical",
            value: string,
        ) => void;
        updateHandoffStaff: (
            shift: "day" | "night",
            type: "delivers" | "receives" | "tens",
            staffList: string[],
        ) => void;
        updateMedicalHandoffDoctor: (doctorName: string) => Promise<void>;
        updateMedicalSignature: (
            doctorName: string,
            scope?: MedicalHandoffScope,
        ) => Promise<void>;
        updateMedicalSpecialtyNote: (
            specialty: MedicalSpecialty,
            value: string,
            actor: Partial<MedicalHandoffActor>,
        ) => Promise<void>;
        updateNurse: (shift: "day" | "night", index: number, name: string) => void;
        updatePatient: (
            bedId: string,
            field: keyof PatientData,
            value: PatientFieldValue,
        ) => void;
        updatePatientMultiple: (
            bedId: string,
            fields: Partial<PatientData>,
        ) => void;
        updateTens: (shift: "day" | "night", index: number, name: string) => void;
        updateTransfer: UpdateTransferAction;
        validateRecordSchema: (
            record: DailyRecord,
        ) => { errors: ZodIssue[]; isValid: boolean };
    }
